
# To Create a Service using Yaml - NodePort ( 30000 - 32767 ) :

apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  type: NodePort
  selector:
    app: mario
  ports:
  - port: 80
    targetPort: 80
    nodePort: 30008

# To Create a Service using Yaml - ClusterIp :

apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  selector:
    app: nginx-app
  ports:
    - port: 80
	  targetPort: 80

# To Create a Service using Yaml - ClusterIp :
	  
apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  type: LoadBalancer
  selector:
    app: nginx-app
  ports:
  - port: 80
    targetPort: 80
	  
# Commands :

kubectl apply -f service.yaml
