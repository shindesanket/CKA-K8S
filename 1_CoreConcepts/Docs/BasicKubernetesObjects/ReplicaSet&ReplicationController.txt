
# Create ReplicaSet using YAML : replicaset.yaml

apiVersion: apps/v1
kind: ReplicaSet
metadata:
  name: frontend
  labels:
    app: guestbook
    tier: frontend
spec:
  # modify replicas according to your case
  replicas: 3
  selector:
    matchLabels:
      tier: frontend
  template:
    metadata:
      labels:
        tier: frontend
    spec:
      containers:
      - name: php-redis
        image: gcr.io/google_samples/gb-frontend:v3
		
# Commands related to ReplicaSet

kubectl get pods -o wide

# Try deleting any pod from replicaset using following command :

kubectl delete pod <pod-id>

# Observe the behaviour that replicaset creates a new pod to achieve desired state configuration

kubectl apply -f replicaset.yaml

kubectl get replicaset

kubectl delete replicaset frontend

kubectl apply -f replicaset.yaml

kubectl scale replicas=6 -f replicaset frontend
